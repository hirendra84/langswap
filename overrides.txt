# Dependency overrides for the Docker image.
#
# vllm-omni's voice cloning requires transformers>=5.3.0, so we pin to the
# lock-file version 5.9.0.  qwen-asr / qwen-tts hard-pin transformers==4.57.x but
# are installed with --no-deps and run fine on 5.x via the in-process compat
# shims, so they do not constrain this.
#
# vllm is also pinned to the lock version: requirements.txt leaves it unpinned,
# so the latest (0.22.0) gets installed, but vllm-omni 0.20.0 expects a matching
# minor and warns of likely compatibility issues.  0.21.0 is the validated pair.
# See requirements.lock.txt.
transformers==5.9.0
vllm==0.21.0
